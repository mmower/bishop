def tokenize( item, stop_words=[] )
  item.split( /\s+/ ).map do |i|
    i.split( /\-/ ).map { |token| token.downcase.gsub( /\W/, "" ) }.join( "-" )
  end.reject { |t| t == "" || t == "-" || stop_words.detect { |w| w == t } }
end
